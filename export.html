  // ==== XỬ LÝ EXPORT PDF ==== //
  // Hàm bóc LaTeX từ HTML MathJax
  function extractLatexFromHTML(html) {
    if (!html) return null;
    const match = html.match(/]+type=["']math\/tex["'][^>]*>([\s\S]+?)<\/script>/);
    if (match) return match[1].trim();
    return null;
  }

  // Chuyển LaTeX thành SVG DataURL dùng MathJax
  async function latexToSVGDataURL(latex) {
    if (!latex) return null;
    await MathJax.startup.promise; // Đảm bảo MathJax đã sẵn sàng
    const svgNode = MathJax.tex2svg(latex, { display: true });
    const svg = svgNode.querySelector('svg');
    if (!svg) return null;
    svg.setAttribute("xmlns", "http://www.w3.org/2000/svg");
    svg.style.margin = "6px 0"; svg.style.background = "#fff";
    const svgData = new XMLSerializer().serializeToString(svg);
    const svg64 = btoa(unescape(encodeURIComponent(svgData)));
    return 'data:image/svg+xml;base64,' + svg64;
  }

  // Hàm trả về node cho pdfmake: nếu có LaTeX thì là ảnh, không thì là text
  async function htmlOrLatexToPDFNode(str) {
    if (!str) return '';
    const latex = extractLatexFromHTML(str);
    if (latex) {
      try {
        const imgData = await latexToSVGDataURL(latex);
        if (typeof imgData === 'string' && imgData.startsWith('data:image/svg+xml;base64,')) {
          return { image: imgData, width: 200, alignment: "left", margin: [0,4,0,4] };
        }
      } catch (e) {}
    }
    // Nếu không có LaTeX hoặc lỗi, trả về text thuần đã loại bỏ HTML
    const tempDiv = document.createElement('div');
    tempDiv.innerHTML = str;
    return tempDiv.textContent?.trim() || '';
  }

  // Hàm xuất PDF
  async function exportPDF() {
    const body = [
      [
        { text: "STT", style: "tableHeader", alignment: "center" },
        { text: "Câu hỏi", style: "tableHeader", alignment: "center" },
        { text: "Đáp án đúng", style: "tableHeader", alignment: "center" }
      ]
    ];

    for (let i = 0; i < DATA.length; ++i) {
      const q = DATA[i];
      const qNode = await htmlOrLatexToPDFNode(q.noi_dung);
      const answerHTML = q.lua_chon ? q.lua_chon[q.dap_an_dung] : '';
      const answerNode = await htmlOrLatexToPDFNode(answerHTML);
      body.push([
        { text: (i+1).toString(), alignment: "center" },
        qNode,
        answerNode
      ]);
    }

    const docDefinition = {
      content: [
        { text: "Đề cương môn học - Export PDF", style: "header", margin: [0, 0, 0, 12] },
        {
          table: {
            headerRows: 1,
            widths: ["8%", "52%", "40%"],
            body: body
          }
        }
      ],
      styles: {
        header: { fontSize: 18, bold: true, alignment: "center" },
        tableHeader: { bold: true, fillColor: "#eeeeee" }
      },
      defaultStyle: { fontSize: 12 }
    };

    pdfMake.createPdf(docDefinition).download('DeCuong_ToanHoc.pdf');
  }
